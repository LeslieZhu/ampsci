59  // look for wildcard:
60  const auto wc = std::find(s2.cbegin(), s2.cend(), '*');
61  if (wc == s2.cend())
62  return s1 == s2;
63 
64  const auto pos_wc = std::size_t(std::distance(s2.cbegin(), wc));
65 
66  const auto s1_front = s1.substr(0, pos_wc);
67  const auto s2_front = s2.substr(0, pos_wc);
68 
69  // number of characters following the '*'
70  const auto len_back = std::size_t(std::distance(wc + 1, s2.cend()));
71 
72  const auto pos_1_back = s1.length() > len_back ? s1.length() - len_back : 0;
73  const auto s1_back = s1.substr(pos_1_back, std::string::npos);
74  const auto s2_back = s2.substr(pos_wc + 1, std::string::npos);
75
